var sum, I, lihai, a5;   { 声明变量sum用于存储总和，变量i用于循环计数 }

    begin
        { 打印整数的代码将在这里，这通常是系统提供的 }

begin  { 主程序开始 }
    sum := 0;  { 初始化总和为0 }
    i := 1;    { 初始化循环计数器为1 }

    while i <= n do  { 当i小于等于n时循环 }
